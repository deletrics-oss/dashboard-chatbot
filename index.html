const express = require('express'); const http = require('http'); const { Server } = require('socket.io'); const { Client } = require('whatsapp-web.js'); const qrcode = require('qrcode'); const { storage } = require('./storage'); // Adicionado para usar o storage const app = express(); const server = http.createServer(app); const io = new Server(server); const whatsappClient = new Client({ puppeteer: { headless: true } }); let qrCodeData = null; whatsappClient.on('qr', (qr) => { console.log('Gerando QR Code...'); qrcode.toDataURL(qr, (err, url) => { if (err) { console.error('Erro ao gerar QR Code:', err); io.emit('status', 'Erro ao gerar QR'); return; } qrCodeData = url; io.emit('qrCode', qrCodeData); console.log('QR Code gerado e enviado'); }); }); whatsappClient.on('ready', () => { console.log('Bot conectado!'); io.emit('status', 'Conectado'); }); whatsappClient.on('message', async (msg) => { if (msg.body === 'ping') msg.reply('pong'); await storage.createMessage({ phoneNumber: msg.from, content: msg.body, isFromBot: false }); io.emit('newMessage', { from: msg.from, content: msg.body }); }); whatsappClient.on('disconnected', (reason) => { console.log('Desconectado:', reason); io.emit('status', 'Desconectado'); qrCodeData = null; }); whatsappClient.initialize().catch(err => console.error('Erro ao inicializar:', err)); app.use(express.static('public')); app.get('/', (req, res) => { res.sendFile(__dirname + '/public/index.html', (err) => { if (err) res.status(500).send('Erro: index.html não encontrado.'); }); }); server.listen(3000, () => { console.log('Servidor rodando em http://162.245.191.70:3000'); }).on('error', (err) => { console.error('Erro ao iniciar servidor:', err); });
